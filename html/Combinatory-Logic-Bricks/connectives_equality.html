Theory connectives_equality

section ‹Logical Connectives using Primitive Equality›
text ‹Via positiva: equality (notation: ‹𝒬›, infix ‹=›) is all you can tell.›

theory connectives_equality
  imports logic_bridge
begin

subsection ‹Basic Connectives›

subsubsection ‹Verum›
text ‹Since any function is self-identical, the following serves as definition of verum/true.›
lemma true_defQ: "𝒯 = 𝒬 𝒬 𝒬" by simp
lemma "𝒯 = (𝒬 = 𝒬)" by simp

subsubsection ‹Identity (for booleans)›
text ‹In fact, the identity function (for booleans) is also definable from equality alone.›
lemma id_defQ: "❙I = 𝒬 𝒯" unfolding comb_defs by auto
lemma "❙I = 𝒬 (𝒬 𝒬 𝒬)" unfolding comb_defs by auto (*expanded*)

subsubsection ‹Falsum›
text ‹Asserting that two different functions are equal is a good way to encode falsum.›
lemma false_defQ: "ℱ = 𝒬 ❙I (❙K 𝒯)" unfolding comb_defs by metis
lemma "ℱ = (❙I = ❙K 𝒯)" unfolding comb_defs by metis
lemma "ℱ = 𝒬(𝒬(𝒬 𝒬 𝒬))(❙K(𝒬 𝒬 𝒬))" unfolding comb_defs by metis  (*expanded*)

subsubsection ‹Negation›
text ‹We can negate a proposition P by asserting that 'P is absurd' (i.e. P is equal to falsum).›
lemma not_defQ: "(¬) = 𝒬 ℱ" unfolding comb_defs by auto
lemma "(¬) = (λP. P = ℱ)" by simp
lemma "(¬) = 𝒬(𝒬(𝒬(𝒬 𝒬 𝒬))(❙K(𝒬 𝒬 𝒬)))" unfolding comb_defs apply (rule ext) by metis (*expanded*)

subsubsection ‹Disequality›
text ‹Using negation we can define disequality for any type (not only boolean).›
lemma diseq_defQ: "𝒟 = (¬) ∘⇩2 𝒬" unfolding comb_defs by simp
lemma "𝒟 = (λA B. ¬(A = B))" by simp

named_theorems eq_defs
declare true_defQ [eq_defs] id_defQ [eq_defs] 
        false_defQ [eq_defs] not_defQ [eq_defs] diseq_defQ [eq_defs]

end
