Theory connectives_disequality

section ‹Logical Connectives using Primitive Disequality›
text ‹Via negativa: disequality (notation: ‹𝒟›, infix ‹≠›) is all you can tell.›

theory connectives_disequality
  imports logic_bridge
begin

subsection ‹Basic Connectives›

subsubsection ‹Falsum›
text ‹Since no function is non-self-identical, the following serves as definition of falsum/false.›
lemma false_defD: "ℱ = 𝒟 𝒟 𝒟" by simp 
lemma "ℱ = (𝒟 ≠ 𝒟)" by simp

subsubsection ‹Identity (for booleans)›
text ‹In fact, the identity function (for booleans) is also definable from disequality alone.›
lemma id_defD: "❙I = 𝒟 ℱ" unfolding comb_defs by simp
lemma "❙I = 𝒟 (𝒟 𝒟 𝒟)" unfolding comb_defs by simp (*expanded*)

subsubsection ‹Verum›
text ‹Asserting that two different functions are different is a good way to encode verum.›
lemma true_defD: "𝒯 = 𝒟 ❙I (❙K ℱ)" unfolding comb_defs by metis
lemma "𝒯 = (❙I ≠ ❙K ℱ)" unfolding comb_defs by metis
lemma "𝒯 = 𝒟(𝒟(𝒟 𝒟 𝒟))(❙K(𝒟 𝒟 𝒟))" unfolding comb_defs by metis (*expanded*)

subsubsection ‹Negation›
text ‹We can negate a proposition P by asserting that "P is not true" (i.e. P is not equal to verum).›
lemma not_defD: "(¬) = 𝒟 𝒯" unfolding comb_defs by simp
lemma "(¬) = (λP. P ≠ 𝒯)" by simp
lemma "(¬) = 𝒟(𝒟(𝒟(𝒟 𝒟 𝒟))(❙K(𝒟 𝒟 𝒟)))" unfolding comb_defs by metis (*expanded*)

subsubsection ‹Equality›
text ‹Using negation we can define equality for any type (not only boolean).›
lemma eq_defD: "𝒬 = (¬) ∘⇩2 𝒟" unfolding comb_defs by simp
lemma "𝒬 = (λA B. ¬(A ≠ B))"  by simp

named_theorems diseq_defs
declare false_defD [diseq_defs] id_defD [diseq_defs] 
        true_defD [diseq_defs] not_defD [diseq_defs] eq_defD [diseq_defs]

end
