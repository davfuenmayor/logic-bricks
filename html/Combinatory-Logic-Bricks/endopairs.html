Theory endopairs

section ‚ÄπEndopairs‚Ä∫

theory endopairs (* A basic theory of endopairs *)
  imports logic_bridge
begin

named_theorems endopair_defs and endopair_simps

subsection ‚ÄπDefinitions‚Ä∫

text ‚ÄπTerm constructor: making an endopair out of two given objects.‚Ä∫
definition mkEndopair::"'a ‚áí 'a ‚áí EPair('a)" ("<_,_>") 
  where "mkEndopair ‚â° ‚ùôL If"

declare mkEndopair_def[endopair_defs]

text ‚ÄπWith syntactic sugar the above definition looks like:‚Ä∫
lemma "<x,y> = (Œªb. if b then x else y)"
  unfolding endopair_defs comb_defs ..

text ‚ÄπUnder the hood, the term constructor ‚ÄπmkEndopair‚Ä∫ is built in terms of definite descriptions.‚Ä∫
lemma mkEndopair_def2: "<x,y> = (Œªb. Œπ z. (b ‚Üí z = x) ‚àß (¬¨b ‚Üí z = y))" 
  unfolding endopair_defs comb_defs unfolding If_def by simp

text ‚ÄπIncidentally, (endo)pairs of booleans have an alternative, simpler representation.‚Ä∫
lemma mkEndopair_bool_simp: "<x,y> = (Œªb. (b ‚àß x) ‚à® (¬¨b ‚àß y))" 
  apply(rule ext) unfolding endopair_defs comb_defs by simp

text ‚ÄπComponentwise equality comparison between endopairs (added as convenient simplification rule).‚Ä∫
lemma mkEndopair_equ_simp: "(<x‚á©1,x‚á©2> = <y‚á©1,y‚á©2>) = (x‚á©1 = y‚á©1 ‚àß x‚á©2 = y‚á©2)"
  unfolding endopair_defs comb_defs by metis

text ‚ÄπWe conveniently add the previous lemmata as a simplification rules.‚Ä∫
declare mkEndopair_bool_simp[endopair_simps] and mkEndopair_equ_simp[endopair_simps]

text ‚ÄπNow, observe that:‚Ä∫
lemma "<x,y> ùíØ = x" 
  unfolding endopair_defs comb_defs by simp
lemma "<x,y> ‚Ñ± = y" 
  unfolding endopair_defs comb_defs by simp

text ‚ÄπThis motivates the introduction of the following projection/extraction functions.‚Ä∫
definition proj1::"EPair('a) ‚áí 'a" ("œÄ‚á©1")
  where "œÄ‚á©1 ‚â° ‚ùôT ùíØ"
definition proj2::"EPair('a) ‚áí 'a" ("œÄ‚á©2")
  where "œÄ‚á©2 ‚â° ‚ùôT ‚Ñ±"

declare proj1_def[endopair_defs] proj2_def[endopair_defs]

lemma "œÄ‚á©1 = (ŒªP. P ùíØ)"   (*sanity check*)
  unfolding endopair_defs comb_defs ..
lemma "œÄ‚á©2 = (ŒªP. P ‚Ñ±)"   (*sanity check*)
  unfolding endopair_defs comb_defs ..

text ‚ÄπThe following lemmata (aka. "product laws") verify that the previous definitions work as intended.‚Ä∫
lemma proj1_simp: "œÄ‚á©1 <x,y> = x" 
  unfolding endopair_defs comb_defs by simp
lemma proj2_simp: "œÄ‚á©2 <x,y> = y" 
  unfolding endopair_defs comb_defs by simp
lemma mkEndopair_simp: "<œÄ‚á©1 P, œÄ‚á©2 P> = P" 
  unfolding endopair_defs comb_defs apply(rule ext) by simp

text ‚ÄπWe conveniently add them as simplification rules.‚Ä∫
declare proj1_simp[endopair_simps] proj2_simp[endopair_simps] mkEndopair_simp[endopair_simps]

text ‚ÄπLet's now add a useful "swap" (endo)operation on endopairs.‚Ä∫
definition swap::"EOp(EPair('a))" 
  where "swap ‚â° ‚ùôC ‚ùôB (¬¨)"

declare swap_def[endopair_defs]

lemma "swap p = p ‚àò (¬¨)" unfolding endopair_defs comb_defs ..
lemma "swap p = (Œªb. p (¬¨b))" unfolding endopair_defs comb_defs ..

text ‚ÄπWe conveniently prove and add some useful simplification rules.‚Ä∫
lemma swap_simp1: "swap <a,b> = <b,a>" 
  unfolding endopair_defs comb_defs by auto
lemma swap_simp2: "<œÄ‚á©2 p, œÄ‚á©1 p> = swap p" 
  unfolding endopair_defs comb_defs by auto

declare swap_simp1[endopair_simps] swap_simp2[endopair_simps]


subsection ‚ÄπCurrying‚Ä∫

text ‚ÄπThe morphisms that convert between unary operations on endopairs and (curried) binary operations.‚Ä∫
definition curry::"Op(EPair('a),'b) ‚áí Op‚á©2('a,'b)" ("‚åä_‚åã")
  where "curry ‚â° ‚ùôC ‚ùôB‚á©2 mkEndopair"
definition uncurry::"Op‚á©2('a,'b) ‚áí Op(EPair('a),'b)" ("‚åà_‚åâ")
  where "uncurry ‚â° ‚ùôL ‚ùôŒ¶‚á©2‚á©1 œÄ‚á©1 œÄ‚á©2"

declare curry_def[endopair_defs] uncurry_def[endopair_defs]

text ‚ÄπSome sanity checks:‚Ä∫
lemma "curry f = ‚ùôB‚á©2 f mkEndopair" unfolding endopair_defs comb_defs ..
lemma "curry f = (Œªx y. f <x,y>)" unfolding endopair_defs comb_defs ..
lemma "uncurry f = ‚ùôŒ¶‚á©2‚á©1 f œÄ‚á©1 œÄ‚á©2" unfolding endopair_defs comb_defs ..
lemma "uncurry f = (ŒªP. f (œÄ‚á©1 P) (œÄ‚á©2 P))" unfolding endopair_defs comb_defs ..

text ‚ÄπBoth morphisms constitute an isomorphism (we add them as simplification rules too)‚Ä∫
lemma curry_simp1: "‚åä‚åàf‚åâ‚åã = f" 
  unfolding curry_def uncurry_def comb_defs unfolding endopair_simps ..
lemma curry_simp2: "‚åà‚åäf‚åã‚åâ = f"
  unfolding curry_def uncurry_def comb_defs unfolding endopair_simps ..

declare curry_simp1[endopair_simps] curry_simp2[endopair_simps]

end
