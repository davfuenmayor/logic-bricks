subsubsection ‹Reflexivity and Irreflexivity›

text ‹Relations are called reflexive (aka. diagonal) resp. irreflexive (aka. antidiagonal) when they are
 larger than identity/equality resp. smaller than difference/disequality.›

lemma ‹reflexive R   = 𝒬 ⊆⇧r R› unfolding endorel_defs ..
lemma ‹irreflexive R = R ⊆⇧r 𝒟› unfolding endorel_defs ..

lemma irreflexive_compl: "irreflexive R⇧─ = reflexive R" unfolding endorel_defs rel_defs func_defs comb_defs by auto

lemma irreflexive_def2: "irreflexive = ∄ ∘ Δ" unfolding endorel_defs rel_defs func_defs comb_defs by auto
lemma "reflexive   R = (∀a. R a a)" unfolding reflexive_def2 comb_defs ..
lemma "irreflexive R = (∀a. ¬R a a)" unfolding irreflexive_def2 comb_defs by simp

text ‹We can naturally obtain a reflexive resp. irreflexive relations via the following operators.›
definition reflexiveClosure::"ERel('a) ⇒ ERel('a)"
  where "reflexiveClosure    ≡ (∪⇧r) 𝒬"

  where "irreflexiveInterior ≡ (∩⇧r) 𝒟"

lemma "irreflexiveInterior R = (R ∩⇧r 𝒟)" unfolding endorel_defs rel_defs func_defs comb_defs by auto

subsubsection ‹Symmetry, Connectedness, and co.›

text ‹We introduce two ways of "symmetrizing" a given relation R: The symmetric interior and closure operations.
 The intuition is that the symmetric interior/closure of R intersects/merges R with its converse, thus
 generating R's largest/smallest symmetric sub/super-relation.›
definition symmetricInterior::"ERel('a) ⇒ ERel('a)"
  where "symmetricInterior ≡ ❙S (∩⇧r) ⌣" ― ‹aka. symmetric part of R›
definition symmetricClosure::"ERel('a) ⇒ ERel('a)"
  where "symmetricClosure  ≡ ❙S (∪⇧r) ⌣"

declare symmetricInterior_def[endorel_defs] symmetricClosure_def[endorel_defs]

lemma "symmetricInterior R = R ∩⇧r (R⇧⌣)" unfolding endorel_defs comb_defs ..
lemma "symmetricClosure R = R ∪⇧r (R⇧⌣)" unfolding endorel_defs comb_defs ..

lemma "symmetricInterior R = (λx y. R x y ∧ R y x)" unfolding endorel_defs rel_defs func_defs comb_defs ..
lemma "symmetricClosure R  = (λx y. R x y ∨ R y x)" unfolding endorel_defs rel_defs func_defs comb_defs ..

lemma symmetricInterior_def2: "symmetricInterior = ❙W ∘ interval" unfolding endorel_defs rel_defs func_defs comb_defs ..
lemma symmetricClosure_def2:  "symmetricClosure  = ❙W ∘ dualInterval" unfolding endorel_defs rel_defs func_defs comb_defs ..

lemma "symmetricInterior R a = (λx. R-interval a a x)" unfolding endorel_defs rel_defs func_defs comb_defs ..
lemma "symmetricClosure R a = (λx. R-dualInterval a a x)" unfolding endorel_defs rel_defs func_defs comb_defs ..

subsubsection ‹Limit-completeness›
text ‹Limit-completeness is an important property of endorelations (orderings in particular). Famously,
 this is the property that characterizes the ordering of real numbers (in contrast to the rationals).›

text ‹Note that existence of lubs for all sets entails existence of glbs for all sets (and viceversa).›
lemma "∀S. ∃(R-lub S) ⟹ ∀S. ∃(R-glb S)" by (simp add: glb_def3)
lemma "∀S. ∃(R-glb S) ⟹ ∀S. ∃(R-lub S)" by (simp add: lub_def3)

text ‹The above results motivate the following definition: An endorelation R is called limit-complete
 when lubs/glbs (wrt R) exist for every set S (note that they must not be necessarily contained in S).›
definition limitComplete::"Set(ERel('a))"
  where "limitComplete ≡ ∀ ∘ (∃ ∘⇩2 lub)" 

lemma "limitComplete R = (∀S. ∃(R-lub S))" unfolding limitComplete_def comb_defs ..

proposition "limitComplete R ⟹ (R-lub S) ⊆ S" nitpick ― ‹countermodel found› oops

text ‹Transpose/converse definitions.›
lemma limitComplete_def2: "limitComplete =  ∀ ∘ (∃ ∘⇩2 glb)" unfolding limitComplete_def comb_defs by (metis glb_def3 lub_def3)
lemma "limitComplete R = (∀S. ∃(R-glb S))" unfolding limitComplete_def2 comb_defs ..

lemma limitComplete_defT: "limitComplete R⇧⌣ =  limitComplete R"
  unfolding limitComplete_def2 comb_defs by (metis glb_def3 lub_def3 lub_defT)

declare limitComplete_def[endorel_defs]

text ‹The subset and subrelation relations are indeed limit-complete.›
lemma subset_limitComplete: "limitComplete (⊆)" unfolding limitComplete_def comb_defs using bigunion_lub by blast
lemma subrel_limitComplete: "limitComplete (⊆⇧r)" unfolding limitComplete_def comb_defs using bigunionR_lub by blast
