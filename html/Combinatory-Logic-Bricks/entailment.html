Theory entailment

section ‹Entailment and Validity›

theory entailment
  imports adjunctions
begin

subsection ‹Special Case (for Modal Logicians and co.)›

text ‹Modal logics model propositions as sets (of "worlds") and are primarily concerned with "validity" 
 of propositions. We encode below the set of valid (resp. unsatisfiable, satisfiable) propositions.›
definition valid::"Set(Set('a))" ("⊨ _") 
  where "valid ≡ ∀"
definition satisfiable::"Set(Set('a))" ("⊨⇧s⇧a⇧t _")
  where "satisfiable ≡ ∃"
definition unsatisfiable::"Set(Set('a))" ("⊨⇧u⇧n⇧s⇧a⇧t _")
  where "unsatisfiable ≡ ∄"

lemma "⊨ P = (∀w. P w)" unfolding valid_def .. ― ‹"true in all worlds"›
lemma "⊨⇧s⇧a⇧t P = (∃w. P w)" unfolding satisfiable_def .. ― ‹"true in some world"›
lemma "⊨⇧u⇧n⇧s⇧a⇧t P = (¬(∃w. P w))" unfolding unsatisfiable_def empty_def .. ― ‹"true in no world"›

text ‹In modal logic, logical consequence/entailment usually comes in two flavours: "local" and "global".
 The local variant is the default one (i.e. the one employed in most sources). Semantically, it 
 corresponds to the subset relation (assumptions are aggregated using conjunction/intersection).›
abbreviation(input) localEntailment::"ERel(Set('a))" (infixr "⊨⇩l" 99)                   
  where "a ⊨⇩l c ≡ a ⊆ c"
abbreviation(input) localEntailment2::"Set('a) ⇒ ERel(Set('a))" ("_,_ ⊨⇩l _") 
  where "a⇩1, a⇩2 ⊨⇩l c ≡ (a⇩1 ∩ a⇩2) ⊨⇩l c"       ― ‹syntax sugar for two (or more) premises›
― ‹...and so on›

text ‹Clearly, validity is a special case of local entailment.›
lemma "⊨ c ↔ 𝔘 ⊨⇩l c" unfolding valid_def func_defs comb_defs by simp

text ‹In fact, local entailment can also be stated in terms of validity via the so-called "deduction
 (meta-)theorem", which follows as a particular case of the following fact (aka. "residuation law").›
lemma local_residuation: "(⊨⇩l),(⊨⇩l)-ADJ⇩2 (∩) (⇒)" by (simp add: adjunction_lift2 and_impl_adj impl_def inter_def subset_def)
― ‹or, in other words›
lemma "a, b ⊨⇩l c ↔ b ⊨⇩l (a ⇒ c)" using local_residuation unfolding galois2_def relLiftAll_def comb_defs by (metis adjunction_def2)
text ‹Which produces the "deduction meta-theorem" as a particular case (with b = ‹𝔘›).›
lemma DMT: "⊨ a ⇒ c ↔ a ⊨⇩l c " by (simp add: B2_comb_def subset_setdef valid_def)

text ‹Global entailment is sometimes discussed, mostly for theoretical purposes (e.g. in algebraic logic).›
abbreviation(input) globalEntailment::"Rel(Set(Set('a)),Set('a))" (infixr "⊨⇩g" 99)
  where "A ⊨⇩g c ≡ (∀a. A a → ⊨ a) → ⊨ c"

text ‹Again, validity is clearly a special case of global entailment.›
lemma "⊨ c ↔ {𝔘} ⊨⇩g c" by (simp add: K21_comb_def universe_def valid_def)

text ‹Local entailment is stronger than global entailment.›
lemma "a ⊨⇩l c ⟹ {a} ⊨⇩g c" unfolding valid_def func_defs comb_defs by auto
lemma "a,b ⊨⇩l c ⟹ {a,b} ⊨⇩g c" unfolding valid_def func_defs comb_defs by auto
lemma "{a} ⊨⇩g c ⟹ a ⊨⇩l c" nitpick oops ― ‹countermodel›

text ‹The "deduction meta-theorem" does not hold for global entailment.›
lemma "{a} ⊨⇩g c ⟹ (⊨ a ⇒ c)" nitpick oops ― ‹countermodel›
lemma "{a,b} ⊨⇩g c ⟹ {b} ⊨⇩g (a ⇒ c)" nitpick oops ― ‹countermodel›

― ‹Only this kind of "detachment rule" holds.›
lemma "⊨ (a ⇒ c) ⟹ {a} ⊨⇩g c" unfolding valid_def func_defs comb_defs by auto
lemma "{b} ⊨⇩g (a ⇒ c) ⟹ {a,b} ⊨⇩g c" unfolding valid_def func_defs comb_defs by auto


subsection ‹General Case (for Algebraic and Many-valued/Fuzzy Logicians)›

text ‹We introduce an "entailment" operation (for denotations) that corresponds to the semantic counterpart
 of the notion of consequence (for formulas). We refer to the literature on algebraic logic for detailed
 explanations, in particular \<^cite>‹Font2007› for an overview, and references therein.›

text ‹We encode below a general notion of logical entailment as discussed in the algebraic logic literature;
 cf. "ramified matrices" (e.g. \<^cite>‹Wojcicki1970›) and "generalized matrices" (e.g. \<^cite>‹Font2007›).
 Entailment becomes parameterized with a class ‹TT› of "truth-sets". We say that a set of assumptions
 ‹A› entails the conclusion ‹c› iff when all ‹A›s are in ‹T› then ‹c› is in ‹T› too, for all truth-sets ‹T› in ‹TT›.›
definition entailment::"Set(Set('a)) ⇒ SetEOp('a)" ("ℰ")
  where "ℰ TT ≡ λA. λc. ∀T. TT T ⟶ A ⊆ T ⟶ T c"

notation(input) entailment ("[_|_ ⊨ _]")
lemma "[TT| A ⊨ c] = ℰ TT A c" .. 

text ‹Alternative definition: ‹c› is in the intersection of all truth-sets containing ‹A››
lemma entailment_def2: "[TT| A ⊨ c] =  ⋂(TT ∩ (⊆) A) c"
  unfolding entailment_def func_defs comb_defs by (smt (z3))

text ‹It is worth noting that when the class of truth-sets ― ‹TT› is closed under arbitrary intersections 
(aka. "closure system") then entailment becomes a closure (aka. hull) operator.›
lemma entailment_closure: "∀X. X ⊆ TT ⟶ TT (⋂X) ⟹ (⊆)-CLOSURE (ℰ TT)" 
  unfolding entailment_def rel_defs func_defs comb_defs by blast

text ‹One special case of the definition above occurs when ‹TT› is a singleton ‹{T}›. This corresponds to
 the traditional notion of logical consequence associated to "logical matrices" in algebraic logic, and 
 which is characterized by the principle of truth(-value) preservation ("truth-preserving consequence"
 in \<^cite>‹Font2007›). We thus refer to its encoding below as "(truth-)value-preserving entailment".›
definition valueEntailment::"Set('a) ⇒ SetEOp('a)" ("ℰ⇩v")
  where "ℰ⇩v T ≡ ℰ {T}"

notation(input) valueEntailment ("[_|_ ⊨⇩v _]")
lemma "[T| A ⊨⇩v c] = ℰ⇩v T A c" ..

text ‹Alternative definition: if all As are in T (true) then c is also in T (true).›
lemma valueEntailment_def2: "[T| A ⊨⇩v c] = (A ⊆ T ⟶ T c)"
  unfolding entailment_def valueEntailment_def by simp

text ‹Value-preserving entailment is a closure operator too.›
lemma ValueConsequence_closure: "(⊆)-CLOSURE (ℰ⇩v T)" 
  unfolding valueEntailment_def entailment_def rel_defs func_defs comb_defs by blast

text ‹Back to the general notion of entailment, now observe that it satisfies the following properties:›
lemma entailment_prop1: "transitive R ⟹ R-glb A m ⟹ [range R | A ⊨ c] = R m c" 
  unfolding entailment_def endorel_defs rel_defs func_defs comb_defs by blast
lemma entailment_prop2: "preorder R ⟹ [range R | {a} ⊨ c] = R a c"
  unfolding entailment_def endorel_defs rel_defs func_defs comb_defs by metis

text ‹The properties above justify the following special case, in which the class of truth-sets is 
 given as the (functional) range of a relation (qua set-valued function). 
 Following \<^cite>‹Font2007› we speak of "(truth-)degree-preserving" entailment.›
definition degreeEntailment::"ERel('a) ⇒ SetEOp('a)" ("ℰ⇩d")
  where "ℰ⇩d R ≡ ℰ (range R)"

notation(input) degreeEntailment ("[_|_ ⊨⇩d _]")
lemma "[R| A ⊨⇩d c] = ℰ⇩d R A c" ..

text ‹Alternative definitions for transitive relations resp. preorders.›
lemma degreeEntailment_def2: "transitive R ⟹ R-glb A h ⟹ [R| A ⊨⇩d c] = R h c" 
  unfolding degreeEntailment_def entailment_prop1 by (smt (z3))
lemma degreeEntailment_def3: "preorder R ⟹ [R| {a} ⊨⇩d c] = R a c"
  by (simp add: degreeEntailment_def entailment_prop2)

text ‹Degree-preserving entailment is a closure operator.›
lemma degreeEntailment_closure: "(⊆)-CLOSURE (ℰ⇩d R)" 
  unfolding entailment_def degreeEntailment_def unfolding rel_defs func_defs comb_defs by blast

text ‹It is worth mentioning that for semantics based on algebras of sets (e.g. modal algebras/Kripke models)
 the usual notion of logical consequence ("local consequence") corresponds to the "degree-preserving"
 entailment presented here, when instantiated with the subset relation.›
lemma degreeEntailment_local: "[(⊆)| A ⊨⇩d c] = ⋂A ⊨⇩l c" 
  by (metis biginter_glb degreeEntailment_def2 partial_order_def2 subset_partial_order)

text ‹Similarly, the notion of "global consequence" (e.g. in modal logic) corresponds to "value-preserving"
 consequence instantiated with ‹T = {𝔘}› where ‹𝔘› is the universe of all points (or "worlds").›
lemma valueEntailment_global: "[{𝔘}| A ⊨⇩v c] = A ⊨⇩g c" 
  unfolding valueEntailment_def2 valid_def func_defs comb_defs by blast

end
