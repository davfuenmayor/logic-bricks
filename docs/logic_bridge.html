theory logic_bridge 
   imports combinators
begin

subsection ‹Custom Tool Configuration›

declare[[smt_timeout=30]]
(* declare[[show_types]] *)

subsection ‹Custom Type Notation›

subsubsection ‹Basic Types›

text ‹Classical HOL systems come with a built-in boolean type, for which we introduce convenient notation alias.›
type_notation bool ("o")

text ‹The creation of a functional type (starting with a type ‹'a›) can be seen from two complementary perspectives:
 Environmentalization (aka. indexation or contextualization) and valuation (e.g. classification, coloring, etc.).›
type_synonym ('e,'a)Env = "'e ⇒ 'a" ("_-Env'(_')" [1000])
type_synonym ('v,'a)Val = "'a ⇒ 'v" ("_-Val'(_')" [1000])


subsubsection ‹Pairs and Sets›

text ‹Starting with the boolean type, we immediately obtain endopairs resp. sets via indexation resp. valuation.›
type_synonym ('a)EPair = "o-Env('a)" ("EPair'(_')")  ― ‹an endopair is encoded as a boolean-index›
type_synonym ('a)Set = "o-Val('a)" ("Set'(_')")  ― ‹a set is encoded as a boolean-valuation (boolean classifier)›

term "((P :: EPair('a)):: 'a-Val(o)) :: o ⇒ 'a"
term "((S :: Set('a)):: 'a-Env(o)) :: 'a ⇒ o"

text ‹Sets of endopairs correspond to (directed) graphs (which are isomorphic to relations via currying).›
type_synonym ('a)Graph = "Set(EPair('a))" ("Graph'(_')")
term "(G :: Graph('a)) :: (o ⇒ 'a) ⇒ o"

text ‹Spaces (sets of sets) are the playground of mathematicians, so they deserve a special type notation.›
type_synonym ('a)Space = "Set(Set('a))" ("Space'(_')")
term "(S :: Space('a)) :: ('a ⇒ o) ⇒ o"


subsubsection ‹Relations›

text ‹Valuations can be made binary (useful e.g. for classifying pairs of objects or encoding their "distance").›
type_synonym ('v,'a,'b)Val2 = "'a ⇒ 'b ⇒ 'v" ("_-Val⇩2'(_,_')" [1000])

text ‹Binary valuations can also be seen as indexed (unary) valuations.›
term "((F :: 'v-Val⇩2('a,'b)) :: 'a-Env('v-Val('b))) :: 'a ⇒ 'b ⇒ 'v"

text ‹In fact (heterogeneous) relations correspond to o-valued binary functions/valuations.›
type_synonym ('a,'b)Rel = "o-Val⇩2('a,'b)" ("Rel'(_,_')")

text ‹They can also be seen as set-valued functions/valuations or as indexed (families of) sets.›
term "(((R :: Rel('a,'b)) :: Set('b)-Val('a)) :: 'a-Env(Set('b))) :: 'a ⇒ 'b ⇒ o"

text ‹Ternary relations are seen as set-valued binary valuations (partial and non-deterministic binary functions).›
type_synonym ('a,'b,'c)Rel3 = "Set('c)-Val⇩2('a,'b)" ("Rel⇩3'(_,_,_')")

text ‹They can also be seen as indexed binary relations (e.g. an indexed family of programs or (a group of) agents).›
term "((R::Rel⇩3('a,'b,'c)) :: 'a-Env(Rel('b,'c))) :: 'a ⇒ 'b ⇒ 'c ⇒ o"

text ‹In general, we can encode n+1-ary relations as indexed n-ary relations.›
type_synonym ('a,'b,'c,'d)Rel4 = "'a-Env(Rel⇩3('b,'c,'d))" ("Rel⇩4'(_,_,_,_')")

text ‹Convenient notation for the particular case where the relata have all the same type.›
type_synonym ('a)ERel = "Rel('a,'a)" ("ERel'(_')") ― ‹(binary) endorelations›
type_synonym ('a)ERel⇩3 = "Rel⇩3('a,'a,'a)" ("ERel⇩3'(_')") ― ‹ternary endorelations›
type_synonym ('a)ERel⇩4 = "Rel⇩4('a,'a,'a,'a)" ("ERel⇩4'(_')") ― ‹quaternary endorelations›

term "(R :: ERel('a)) :: 'a ⇒ 'a ⇒ o"
term "(R :: ERel⇩3('a)) :: 'a ⇒ 'a ⇒ 'a ⇒ o"
term "(R :: ERel⇩4('a)) :: 'a ⇒ 'a ⇒ 'a ⇒ 'a ⇒ o"


subsubsection ‹Operations›

text ‹As a convenient mathematical abstraction, we introduce the notion of "operation".
In mathematical phraseology, operations are said to "operate" on (one or more) "operands".
Operations can be seen as (curried) functions whose arguments have all the same type.›

text ‹Unary case: (endo)operations just correspond to (endo)functions.›
type_synonym ('a,'b)Op1 = "'a ⇒ 'b" ("Op'(_,_')")
type_synonym ('a)EOp1 = "Op('a,'a)" ("EOp'(_')") ― ‹same as: ‹'a ⇒ 'a› ›

text ‹Binary case: (endo)bi-operations correspond to curried (endo)bi-functions.›
type_synonym ('a,'b)Op2 = "'a ⇒ 'a ⇒ 'b" ("Op⇩2'(_,_')")
type_synonym ('a)EOp2 = "Op⇩2('a,'a)" ("EOp⇩2'(_')") ― ‹same as: ‹'a ⇒ ('a ⇒ 'a)››

text ‹Arbitrary case: generalized (endo)operations correspond to (endo)functions on sets.›
type_synonym ('a,'b)OpG = "Op(Set('a),'b)" ("Op⇩G'(_,_')")
type_synonym ('a)EOpG = "Op⇩G('a,'a)" ("EOp⇩G'(_')") ― ‹same as: ‹Set('a) ⇒ 'a››

text ‹Convenient type aliases for (endo)operations on sets.›
type_synonym ('a,'b)SetOp = "Op(Set('a),Set('b))" ("SetOp'(_,_')")
type_synonym ('a)SetEOp = "SetOp('a,'a)" ("SetEOp'(_')") ― ‹same as: ‹Set('a) ⇒ Set('a)››

text ‹Binary case: (endo)bi-operations correspond to curried (endo)bi-functions›
type_synonym ('a,'b)SetOp2 = "Set('a) ⇒ Set('a) ⇒ Set('b)" ("SetOp⇩2'(_,_')")
type_synonym ('a)SetEOp2 = "SetOp⇩2('a,'a)" ("SetEOp⇩2'(_')") ― ‹same as: ‹Set('a) ⇒ Set('a) ⇒ Set('a)››


subsubsection ‹Products of Boolean Types›

text ‹Now consider the following equivalent type notations.›
term "((S :: Set(o)) :: EPair(o)) :: o ⇒ o"
term "((R :: ERel(o)) :: EOp⇩2(o)) :: o ⇒ (o ⇒ o)"
term "(((S :: Space(o)) :: Graph(o)) :: EOp⇩G(o)) :: (o ⇒ o) ⇒ o"

text ‹We can make good sense of them by considering a new type having four inhabitants.›
type_synonym four = "o ⇒ o" ("oo")
term "((P :: oo) :: EPair(o)) :: Set(o)"

text ‹Using the new type we can seamlessly define types for (endo)quadruples and 4-valued sets.›
type_synonym ('a)EQuad = "oo ⇒ 'a" ("EQuad'(_')")
type_synonym ('a)Set4 = "'a ⇒ oo" ("Set4'(_')")

text ‹The following two types have each 16 elements (we show a bijection between their elements later on).›
type_synonym sixteen  = "o ⇒ oo" ("ooo")   ― ‹‹4^2 = (2^2)^2››
type_synonym sixteen' = "oo ⇒ o" ("ooo''") ― ‹‹2^4 = 2^(2^2)››

text ‹So we can have that the following type notations are in fact identical (not just isomorphic).›
term "(((S :: Set(o)) :: EPair(o)) :: o ⇒ o) :: oo"
term "(((((R :: ERel(o)) :: EOp⇩2(o)) :: EPair(oo)) :: Set4(o)) :: o ⇒ o ⇒ o) :: ooo"
term "((((((S :: Space(o)) :: Graph(o)) :: EOp⇩G(o)) :: Set(oo)) :: EQuad(o)) :: (o ⇒ o) ⇒ o) :: ooo'"

text ‹We can continue producing types (we stop giving them special notation after the magic number 64).›
type_synonym sixtyfour = "oo ⇒ oo" ("oooo") ― ‹‹4^4 = (2^2)^(2^2) = 64››
type_synonym n256   = "o ⇒ ooo" ― ‹‹16^2 = 256››
type_synonym n65536 = "oo ⇒ ooo" ― ‹‹16^4 = 65536››
type_synonym n65536' = "ooo ⇒ o" ― ‹‹2^16 = 65536››
type_synonym n4294967296 = "ooo ⇒ oo" ― ‹‹4^16 = 4294967296››
― ‹and so on...›


text ‹Continuations (with result type ‹'r›) take inputs of type ‹'a››
text ‹Unary case:›
type_synonym ('a,'r)Cont1 = "'r-Val(Op('a,'r))" ("Cont'(_,_')") ― ‹same as: ‹('a ⇒ 'r) ⇒ 'r››
text ‹Binary case:›
type_synonym ('a,'r)Cont2 = "'r-Val(Op⇩2('a,'r))" ("Cont⇩2'(_,_')") ― ‹same as: ‹('a ⇒ 'a ⇒ 'r) ⇒ 'r››


subsection ‹Custom Term Notation›

text ‹Convenient combinator-like symbols ‹𝒬› resp. ‹𝒟› to be used instead of ‹(=)› resp. ‹(≠)›.›
notation HOL.eq ("𝒬") and HOL.not_equal ("𝒟")

(*<*)
(* Removes the (=) resp. (≠) symbols from output (we want to see 𝒬/{_} resp. 𝒟/⦃_⦄ instead) *)
no_notation(output)
  HOL.eq (infix "=" 50) and HOL.not_equal (infix "≠" 50)
notation (output)
  HOL.eq  ("(_ =/ _)" [51, 51] 50) and HOL.not_equal  ("(_ ≠/ _)" [51, 51] 50)
(*>*)

text ‹Alternative (more concise) notation for boolean constants.›
notation HOL.True ("𝒯") and HOL.False ("ℱ")

text ‹Add (binder) notation for indefinite descriptions (aka. Hilbert's epsilon or choice operator).›
notation Hilbert_Choice.Eps ("ε") and Hilbert_Choice.Eps (binder "ε" 10)

text ‹Introduce a convenient "dual" to Hilbert's epsilon operator (adds variable-binding notation).›
definition Delta ("δ")
  where "δ ≡ λA. ε (λx. ¬A x)"
notation Delta (binder "δ" 10) 

text ‹Add (binder) notation for definite descriptions (incl. binder notation).›
notation HOL.The ("ι") and HOL.The (binder "ι" 10)

(*<*)(*Sanity checks*)
lemma "(ε x. A x) = (SOME x. A x)" ..
lemma "(δ x. A x) = (SOME x. ¬A x)" unfolding Delta_def ..
lemma "(ι x. A x) = (THE x. A x)" ..
(*>*)

text ‹We introduce (pedagogically convenient) notation for HOL logical constants.›
notation HOL.All ("∀") 
notation HOL.Ex  ("∃")
abbreviation Empty ("∄")
  where "∄A ≡ ¬∃A"                           

notation HOL.implies (infixr "→" 25)  ― ‹convenient alternative notation›
notation HOL.iff (infixr "↔" 25) ― ‹convenient alternative notation›

text ‹Add convenient logical connectives.›
abbreviation(input) seilpmi (infixl "←" 25) ― ‹reversed implication›
  where "A ← B ≡ B → A"
abbreviation(input) excludes (infixl "↽" 25) ― ‹aka. co-implication›
  where "A ↽ B ≡ A ∧ ¬B"
abbreviation(input) sedulcxe (infixr "⇀" 25) ― ‹aka. dual-implication›
  where "A ⇀ B ≡ B ↽ A"
abbreviation(input) xor (infix "⇌" 25) ― ‹aka. symmetric difference›
  where "A ⇌ B ≡ A ≠ B"
abbreviation(input) nand (infix "↑" 35) ― ‹aka. Sheffer stroke›
  where "A ↑ B ≡ ¬(A ∧ B)"
abbreviation(input) nor (infix "↓" 30) ― ‹aka. Peirce arrow or Quine dagger›
  where "A ↓ B ≡ ¬(A ∨ B)"

text ‹Check relationships›
lemma disj_impl: "(A ∨ B) = ((A → B) → B)" by auto
lemma conj_excl: "(A ∧ B) = ((A ⇀ B) ⇀ B)" by auto
lemma xor_excl: "(A ⇌ B) = (A ↽ B) ∨ (A ⇀ B)" by auto

(*<*)(*Reintroduces notation for negation to allow for conveniently referring to it as '¬' *)
notation(input) HOL.Not ("¬" 40)
notation(output) HOL.Not ("¬_" [40] 40)
