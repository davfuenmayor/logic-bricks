Theory spaces

section ‹Spaces›
text ‹Spaces are sets of sets (of ... "points"). They are the main playground of mathematicians.›

theory spaces (* A theory of spaces qua sets of sets *)
imports endorelations
begin

named_theorems space_defs

subsection ‹Spaces as Quantifiers and co.›

text ‹Quantifiers are particular kinds of spaces.›
term "∀ :: Space('a)" ― ‹‹∀› is the space that contains only the universe›
lemma All_simp1:"{𝔘} = ∀" unfolding func_defs comb_defs by auto
lemma All_simp2:"(⊆) 𝔘 = ∀" unfolding func_defs comb_defs by simp

term "∃ :: Space('a)" ― ‹‹∃› is the space that contains all but the empty set›
lemma Ex_simp1: "⦃∅⦄ = ∃" unfolding Ex_def func_defs comb_defs by auto 
lemma Ex_simp2: "(⊇) ∅ = ∄" unfolding func_defs unfolding comb_defs by simp

term "∄ :: Space('a)"  ― ‹‹∄› is the space that contains only the empty set›
lemma nonEx_simp: "{∅} = ∄" unfolding Ex_def func_defs comb_defs by auto 

text ‹In general, any property of sets corresponds to a space. For instance:›
term "unique :: Space('a)" ― ‹unique is the space that contains all and only univalent sets (having at most one element)›
term "∃! :: Space('a)" ― ‹‹∃!› is the space that contains all and only singleton sets›

lemma unique_def2: "unique = ∄ ∪ ∃!" unfolding func_defs comb_defs by auto
lemma singleton_def2: "∃! = ∃ ∩ unique" unfolding func_defs comb_defs by metis
lemma singleton_def3: "∃!A = (∃a. A = {a})" unfolding singleton_def2 func_defs comb_defs by auto

text ‹Further convenient instances of spaces.›
definition upair::"Space('a)" ("∃⇩≤⇩2") ― ‹‹∃⇩≤⇩2› contains the unordered pairs (sets with one or two elements)›
  where ‹∃⇩≤⇩2 ≡ ∃⇧2 ∘ (❙Φ⇩2⇩1 (∩⇧r) (❙W (×)) (❙R ❙E (❙Ψ⇩2 (∪) 𝒬) (⊆)))›
definition doubleton::"Space('a)" ("∃!⇩2") ― ‹‹∃!⇩2› contains the doubletons (sets with two (different) elements)›
  where ‹∃!⇩2 ≡ ∃⇩≤⇩2 ∖ ∃!›

declare unique_def[space_defs] singleton_def[space_defs] doubleton_def[space_defs] upair_def[space_defs] 

lemma "∃⇩≤⇩2A = ∃⇧2((A × A) ∩⇧r (λx y. A ⊆ {x,y}))" unfolding space_defs comb_defs ..
lemma doubleton_def2: "∃⇩≤⇩2A = (∃x y. A x ∧ A y ∧ (∀z. A z → (z = x ∨ z = y)))" unfolding space_defs rel_defs func_defs comb_defs by auto

lemma ‹∃!⇩2 A = (∃x y. x ≠ y ∧ A x ∧ A y ∧ (∀z. A z → (z = x ∨ z = y)))› unfolding space_defs rel_defs func_defs comb_defs by blast
lemma upair_def2: "∃⇩≤⇩2 = ∃! ∪ ∃!⇩2" unfolding space_defs func_defs rel_defs comb_defs by blast

lemma doubleton_def3: "∃!⇩2A = (∃a b. a ≠ b ∧ A = {a,b})" unfolding space_defs rel_defs func_defs comb_defs by blast
lemma upair_def3: "∃⇩≤⇩2A = (∃a b. A = {a,b})" unfolding space_defs rel_defs func_defs comb_defs by metis


text ‹Convenient abbreviation for sets that have 2 or more elements.›
abbreviation(input) nonUnique::"Space('a)" ("∃⇩≥⇩2")
  where "∃⇩≥⇩2A ≡ ¬(unique A)"

text ‹Sets, in general, are the bigunions of their contained singletons.›
lemma singleton_gen: "S = ⋃(℘S ∩ ∃!)" unfolding singleton_def3 func_defs comb_defs by metis
text ‹Sets with more than one element are the bigunions of their contained doubletons.›
lemma doubleton_gen: "∃⇩≥⇩2 S ⟹ S = ⋃(℘S ∩ ∃!⇩2)"  unfolding doubleton_def3 func_defs comb_defs apply auto apply (rule ext)+ by (smt (verit, best))
text ‹Sets, in general, are the bigunions of their contained unordered pairs.›
lemma upair_gen: "S = ⋃(℘S ∩ ∃⇩≤⇩2)" unfolding upair_def2 singleton_def3 doubleton_def3 unfolding func_defs comb_defs by metis

text ‹Some useful equations:›
lemma singleton_prop: "(∀D. D ⊆ S → ∃!D → P D) = (∀x. S x → P {x})"
  unfolding singleton_def3 unfolding func_defs comb_defs by (metis (full_types))
lemma doubleton_prop: "(∀D. D ⊆ S → ∃!⇩2D → P D) = (∀x y. S x ∧ S y → x ≠ y → P {x,y})"
  unfolding doubleton_def3 unfolding func_defs comb_defs by (smt (verit))
lemma upair_prop: "(∀D. D ⊆ S → ∃⇩≤⇩2D → P D) = (∀x y. S x ∧ S y → P {x,y})" 
  unfolding upair_def3 unfolding func_defs comb_defs by (smt (verit, best))

subsubsection ‹Lub- and Glb-related Definitions›

text ‹These definitions generalize the "complete join/meet-semilattice" property (existence of suprema resp. infima).›
definition lubComplete::"ERel('a) ⇒ Space('a)" ("_-lubComplete")
  where "R-lubComplete ≡ ❙Φ⇩2⇩1 (⊆) ℘ ((❙R ❙D (R-lub)) (⊓))" 
definition glbComplete::"ERel('a) ⇒ Space('a)" ("_-glbComplete")
  where "R-glbComplete ≡ ❙Φ⇩2⇩1 (⊆) ℘ ((❙R ❙D (R-glb)) (⊓))" ― ‹all of S-subsets have a glb (wrt R) in S›

declare lubComplete_def[space_defs] glbComplete_def[space_defs]

text ‹All of S-subsets have a lub (wrt R) in S.›
lemma "R-lubComplete = (λS. ℘ S ⊆ ((❙R ❙D (R-lub)) (⊓) S))" unfolding space_defs comb_defs ..
lemma lubComplete_def2: "R-lubComplete = (λS. ∀D. D ⊆ S ⟶ (R-lub D ⊓ S))" unfolding space_defs rel_defs func_defs comb_defs by blast

text ‹All of S-subsets have a glb (wrt R) in S.›
lemma "R-glbComplete = (λS. ℘ S ⊆ ((❙R ❙D (R-glb)) (⊓) S))" unfolding space_defs comb_defs ..
lemma glbComplete_def2: "R-glbComplete = (λS. ∀D. D ⊆ S ⟶ (R-glb D ⊓ S))" unfolding space_defs rel_defs func_defs comb_defs by blast

lemma lubComplete_defT: "R-lubComplete = R⇧⌣-glbComplete" unfolding space_defs endorel_defs rel_defs comb_defs ..
lemma glbComplete_defT: "R-glbComplete = R⇧⌣-lubComplete" unfolding space_defs endorel_defs rel_defs comb_defs ..

text ‹Limit-completeness of a relation can be expressed in terms of either lub- or glb-completeness.›
lemma limitComplete_def3: "limitComplete R = R-lubComplete 𝔘"
  unfolding space_defs endorel_defs func_defs comb_defs by simp
lemma limitComplete_def4: "limitComplete R = R-glbComplete 𝔘"
  unfolding limitComplete_def2 space_defs func_defs comb_defs by simp

text ‹Note that lub/glb-completeness is neither monotonic nor antitonic, for instance:›
proposition "A ⊆ B ⟹ R-lubComplete A ⟹ R-lubComplete B" nitpick ― ‹countermodel found› oops
proposition "A ⊆ B ⟹ R-lubComplete B ⟹ R-lubComplete A" nitpick ― ‹countermodel found› oops


text ‹The following related propertes correspond to closure under the lub resp. glb set-operation wrt R.›
definition lubClosed::"ERel('a) ⇒ Space('a)" ("_-lubClosed")
  where "R-lubClosed ≡ (R-lub)-closed⇩S" 
definition glbClosed::"ERel('a) ⇒ Space('a)" ("_-glbClosed")
  where "R-glbClosed ≡ (R-glb)-closed⇩S"

declare lubClosed_def[space_defs] glbClosed_def[space_defs]

lemma lubClosed_defT: "R-lubClosed = R⇧⌣-glbClosed" unfolding rightBound_defT space_defs endorel_defs rel_defs comb_defs ..
lemma glbClosed_defT: "R-glbClosed = R⇧⌣-lubClosed" unfolding leftBound_defT space_defs endorel_defs rel_defs comb_defs ..

text ‹Recalling that antisymmetry entails uniqueness of lub/glb (when they exist), we have in fact.›
lemma lubComplete_lubClosed: "antisymmetric R ⟹ R-lubComplete S ⟹ R-lubClosed S" 
  unfolding space_defs endorel_defs rel_defs func_defs comb_defs by metis
lemma glbComplete_glbClosed: "antisymmetric R ⟹ R-glbComplete S ⟹  R-glbClosed S" 
  unfolding space_defs endorel_defs rel_defs func_defs comb_defs by metis

text ‹However, being closed under lub/glb does not entail existence of lub/glb.›
proposition "∃S ⟹ R-lubClosed S ⟹  R-lubComplete S" nitpick ― ‹countermodel found› oops
proposition "∃S ⟹ R-glbClosed S ⟹ R-glbComplete S" nitpick ― ‹countermodel found› oops

text ‹In fact, for limit-complete relations, closure under lub/glb does entail existence of lub/glb.›
lemma lubClosed_lubComplete: "limitComplete R ⟹ R-lubClosed S ⟹ R-lubComplete S" 
  unfolding space_defs endorel_defs func_defs comb_defs by metis
lemma glbClosed_glbComplete: "limitComplete R ⟹ R-glbClosed S ⟹  R-glbComplete S" 
  unfolding limitComplete_def2 space_defs endorel_defs func_defs comb_defs by metis

lemma lubClosed_def2: "antisymmetric R ⟹ limitComplete R ⟹ R-lubComplete = R-lubClosed" 
  using lubClosed_lubComplete lubComplete_lubClosed by blast
lemma glbClosed_def2: "antisymmetric R ⟹ limitComplete R ⟹ R-glbComplete = R-glbClosed" 
  using glbClosed_glbComplete glbComplete_glbClosed by blast


subsubsection ‹Upwards- and Downwards-Closure›

definition upwardsClosed::"ERel('a) ⇒ Space('a)" ("_-upwardsClosed")
  where "R-upwardsClosed ≡ (R-upImage)-closed⇩S"
definition downwardsClosed::"ERel('a) ⇒ Space('a)" ("_-downwardsClosed")
  where "R-downwardsClosed ≡ (R-downImage)-closed⇩S"

declare upwardsClosed_def[space_defs] downwardsClosed_def[space_defs]

lemma upwardsClosed_defT: "R-upwardsClosed = R⇧⌣-downwardsClosed" unfolding space_defs rightImage_defT ..
lemma downwardsClosed_defT: "R-downwardsClosed = R⇧⌣-upwardsClosed" unfolding space_defs leftImage_defT ..

lemma upwardsClosed_def2: "R-upwardsClosed S = (∀x y. R x y ⟶ S x ⟶ S y)"
  unfolding space_defs rel_defs func_defs comb_defs by blast
lemma downwardsClosed_def2: "R-downwardsClosed S = (∀x y. R x y ⟶ S y ⟶ S x)"
  unfolding space_defs rel_defs func_defs comb_defs by blast

lemma upwardsClosed_def3: "skeletal R ⟹ R-upwardsClosed S = (∀D. ∃(R-glb D) ⟶ (R-glb D) ⊆ S  ⟶ D ⊆ S)"
  unfolding space_defs endorel_defs rel_defs func_defs comb_defs apply (rule iffI) apply blast by (smt (verit, del_insts))
lemma downwardsClosed_def3: "skeletal R ⟹ R-downwardsClosed S = (∀D. ∃(R-lub D) ⟶ (R-lub D) ⊆ S  ⟶ D ⊆ S)"
  by (simp add: downwardsClosed_defT lub_defT skeletal_symm upwardsClosed_def3)

lemma upwardsClosed_def4: "skeletal R ⟹ limitComplete R ⟹ R-upwardsClosed S = (∀D. (R-glb D) ⊆ S ⟶ D ⊆ S)"  
  unfolding upwardsClosed_def3 limitComplete_def2 comb_defs by simp
lemma downwardsClosed_def4: "skeletal R ⟹ limitComplete R ⟹ R-downwardsClosed S = (∀D. (R-lub D) ⊆ S ⟶ D ⊆ S)"  
  by (simp add: downwardsClosed_defT limitComplete_defT lub_defT skeletal_symm upwardsClosed_def4)


subsubsection ‹Existence of Greatest- and Least-Elements›

text ‹Another interesting property is existence of greatest resp. least elements.›
definition greatestExist::"ERel('a) ⇒ Space('a)" ("_-greatestExist")
  where "R-greatestExist ≡ ∃ ∘ R-greatest"
definition leastExist::"ERel('a) ⇒ Space('a)" ("_-leastExist")
  where "R-leastExist ≡ ∃ ∘ R-least"

declare greatestExist_def[space_defs] leastExist_def[space_defs]

text ‹In fact, recalling that:›
lemma "R-greatest S = (S ∩ R-upperBound S)" unfolding greatest_def comb_defs ..
lemma "R-least S = (S ∩ R-lowerBound S)" unfolding least_def comb_defs ..

lemma greatestExist_defT: "R-greatestExist = R⇧⌣-leastExist" unfolding rightBound_defT space_defs endorel_defs comb_defs ..
lemma leastExist_defT: "R-leastExist = R⇧⌣-greatestExist" unfolding leftBound_defT space_defs endorel_defs comb_defs ..

text ‹We have that existence of greatest/least elements for S is equivalent to every S-subset having 
 upper/lower bounds (wrt R) in S. (Note that this is a strong variant of up/downwards directedness.)›
lemma greatestExist_def2: "R-greatestExist S = (∀D. D ⊆ S ⟶ ∃(S ∩ R-upperBound D))"
  unfolding space_defs endorel_defs rel_defs func_defs comb_defs by auto
lemma leastExist_def2:    "R-leastExist S    = (∀D. D ⊆ S ⟶ ∃(S ∩ R-lowerBound D))"
  unfolding space_defs endorel_defs rel_defs func_defs comb_defs by auto

text ‹Or, alternatively:›
lemma greatestExist_def3: "R-greatestExist S = (∃S ∧ (∀D. D ⊆ S ⟶ ∃D ⟶ ∃(S ∩ R-upperBound D)))"
  unfolding greatestExist_def2 rel_defs func_defs comb_defs by auto
lemma leastExist_def3:    "R-leastExist S    = (∃S ∧ (∀D. D ⊆ S ⟶ ∃D ⟶ ∃(S ∩ R-lowerBound D)))"
  unfolding leastExist_def2  rel_defs func_defs comb_defs by auto

text ‹In fact, existence of greatest/least-elements is a strictly weaker property than lub/glb-completeness.›
lemma glbComplete_least: "R-glbComplete ⊆ R-leastExist" 
  unfolding space_defs endorel_defs func_defs comb_defs by auto
lemma lubComplete_greatest: "R-lubComplete ⊆ R-greatestExist"
  unfolding space_defs endorel_defs func_defs comb_defs by auto
proposition "R-greatestExist ⊆ R-lubComplete" nitpick ― ‹countermodel found› oops
proposition "R-leastExist ⊆ R-glbComplete" nitpick ― ‹countermodel found› oops

lemma greatestExist_lubClosed: "R-downwardsClosed S ⟹ R-greatestExist S ⟹ R-lubClosed S"
  unfolding space_defs endorel_defs rel_defs func_defs comb_defs by blast
lemma leastExist_glbClosed: "R-upwardsClosed S ⟹ R-leastExist S ⟹ R-glbClosed S"
  unfolding space_defs endorel_defs rel_defs func_defs comb_defs by blast

lemma greatestExist_def4: "limitComplete R ⟹ R-downwardsClosed S ⟹ R-greatestExist S = R-lubClosed S"
  using greatestExist_lubClosed lubClosed_lubComplete lubComplete_greatest unfolding func_defs comb_defs by metis
lemma leastExist_def4: "limitComplete R ⟹ R-upwardsClosed S ⟹ R-leastExist S = R-glbClosed S"
  by (simp add: glbClosed_defT greatestExist_def4 leastExist_defT limitComplete_defT upwardsClosed_defT)


subsubsection ‹Upwards- and Downwards-Directedness›

text ‹The property of a set being "up/downwards directed" wrt. an endorelation:
  Every pair of S-elements has upper/lower-bounds (wrt R) in S.›
definition upwardsDirected::"ERel('a) ⇒ Space('a)" ("_-upwardsDirected")
  where "R-upwardsDirected  ≡ ❙Φ⇩2⇩1 (⊆⇧r) (❙W (×)) (❙R ❙E (❙Ψ⇩2 (∩) R) (⊓))"
definition downwardsDirected::"ERel('a) ⇒ Space('a)" ("_-downwardsDirected")
  where "R-downwardsDirected ≡  ❙Φ⇩2⇩1 (⊆⇧r) (❙W (×)) (❙R ❙E (❙Ψ⇩2 (∩) (❙C R)) (⊓))"

declare upwardsDirected_def[space_defs] downwardsDirected_def[space_defs]

lemma "R-upwardsDirected = (λS. (S × S) ⊆⇧r (λa b. S ⊓ (❙Ψ⇩2 (∩) R a b)))" unfolding space_defs comb_defs ..
lemma "R-upwardsDirected = (λS. (S × S) ⊆⇧r (λa b. S ⊓ (R a ∩ R b) ))" unfolding space_defs comb_defs ..
lemma "R-upwardsDirected = (λS. ∀a b. S a ∧ S b ⟶ (∃c. S c ∧ R a c ∧ R b c))" unfolding space_defs rel_defs func_defs comb_defs ..

lemma "R-downwardsDirected = (λS. (S × S) ⊆⇧r (λa b. S ⊓ (❙Ψ⇩2 (∩) (❙C R) a b)))" unfolding space_defs comb_defs ..
lemma "R-downwardsDirected = (λS. (S × S) ⊆⇧r (λa b. S ⊓ (R⇧⌣ a ∩ R⇧⌣ b)))" unfolding space_defs rel_defs func_defs comb_defs ..
lemma "R-downwardsDirected = (λS. ∀a b. S a ∧ S b ⟶ (∃c. S c ∧ R c a ∧ R c b))" unfolding space_defs rel_defs func_defs comb_defs ..

lemma upwardsDirected_defT: "R-upwardsDirected = R⇧⌣-downwardsDirected" unfolding space_defs rel_defs comb_defs ..
lemma downwardsDirected_defT: "R-downwardsDirected = R⇧⌣-upwardsDirected" unfolding space_defs rel_defs comb_defs ..

text ‹The definition above can be rephrased as:›
lemma upwardsDirected_def2: "R-upwardsDirected S = (∀a b. S a ∧ S b ⟶ ∃(S ∩ R-upperBound {a,b}))" 
  unfolding space_defs rel_defs func_defs comb_defs by metis
lemma downwardsDirected_def2: "R-downwardsDirected S = (∀a b. S a ∧ S b ⟶ ∃(S ∩ R-lowerBound {a,b}))" 
  unfolding space_defs rel_defs func_defs comb_defs by metis
text ‹or, alternatively:›
lemma upwardsDirected_def3:  "R-upwardsDirected S = (∀D. D ⊆ S ⟶ ∃⇩≤⇩2D ⟶ ∃(S ∩ R-upperBound D))"
  unfolding upwardsDirected_def2 unfolding upair_prop ..
lemma downwardsDirected_def3:  "R-downwardsDirected S = (∀D. D ⊆ S ⟶ ∃⇩≤⇩2D ⟶ ∃(S ∩ R-lowerBound D))"
  unfolding downwardsDirected_def2 unfolding upair_prop ..

text ‹Note that up/downwards directedness does not entail non-emptyness of S.›
proposition "R-upwardsDirected S ⟶ ∃S" nitpick  ― ‹countermodel found› oops
proposition "R-downwardsDirected S ⟶ ∃S" nitpick  ― ‹countermodel found› oops


subsubsection ‹Join- and Meet-Closure›

text ‹Convenient abbreviations for joins resp. meets (lub resp. glb of sets with 2 elements).›
abbreviation(input) join ("_-join")
  where "R-join a b ≡ R-lub {a,b}"
abbreviation(input) meet ("_-meet")
  where "R-meet a b ≡ R-glb {a,b}"

text ‹Some platitudes about meets and joins.›
lemma join_prop1:  "R-lowerBound (R-join a b) a" unfolding endorel_defs rel_defs func_defs comb_defs by auto
lemma join_prop2:  "R-lowerBound (R-join a b) b" unfolding endorel_defs rel_defs func_defs comb_defs by auto
lemma meet_prop1:  "R-upperBound (R-meet a b) a" unfolding endorel_defs rel_defs func_defs comb_defs by auto
lemma meet_prop2:  "R-upperBound (R-meet a b) b" unfolding endorel_defs rel_defs func_defs comb_defs by auto

text ‹The following are weaker versions of lub/glb-closure customarily used in the literature.›
definition joinClosed::"ERel('a) ⇒ Space('a)" ("_-joinClosed")
  where "R-joinClosed ≡ ❙Φ⇩2⇩1 (⊆⇧r) (❙W (×)) (❙R ❙E (R-join) ℘)"
definition meetClosed::"ERel('a) ⇒ Space('a)" ("_-meetClosed")
  where "R-meetClosed ≡ ❙Φ⇩2⇩1 (⊆⇧r) (❙W (×)) (❙R ❙E (R-meet) ℘)"

declare joinClosed_def[space_defs] meetClosed_def[space_defs]

lemma "R-joinClosed = (λS. (S × S) ⊆⇧r (❙R ❙E (R-join) ℘ S))" unfolding space_defs comb_defs ..
lemma "R-joinClosed = (λS. (S × S) ⊆⇧r (λa b. R-join a b ⊆ S))" unfolding space_defs comb_defs ..
lemma "R-joinClosed = (λS. ∀a b. S a ∧ S b ⟶ R-join a b ⊆ S)" unfolding space_defs rel_defs func_defs comb_defs ..

lemma "R-meetClosed = (λS. (S × S) ⊆⇧r (❙R ❙E (R-meet) ℘ S))" unfolding space_defs comb_defs ..
lemma "R-meetClosed = (λS. (S × S) ⊆⇧r (λa b. R-meet a b ⊆ S))" unfolding space_defs comb_defs ..
lemma "R-meetClosed = (λS. ∀a b. S a ∧ S b ⟶ R-meet a b ⊆ S)" unfolding space_defs rel_defs func_defs comb_defs ..

lemma joinClosed_defT: "R-joinClosed = R⇧⌣-meetClosed" unfolding space_defs endorel_defs rel_defs func_defs comb_defs ..
lemma meetClosed_defT: "R-meetClosed = R⇧⌣-joinClosed" unfolding space_defs endorel_defs rel_defs func_defs comb_defs ..

lemma joinClosed_def2: "joinClosed R S = (∀p. p ⊆ S → ∃⇩≤⇩2 p → (R-lub p) ⊆ S)" 
  unfolding upair_prop unfolding space_defs endorel_defs rel_defs func_defs comb_defs by auto
lemma meetClosed_def2: "meetClosed R S = (∀p. p ⊆ S → ∃⇩≤⇩2 p → (R-glb p) ⊆ S)" 
  unfolding upair_prop unfolding space_defs endorel_defs rel_defs func_defs comb_defs by auto

lemma joinClosed_upwardsDirected: "limitComplete R ⟹ R-joinClosed S ⟹ R-upwardsDirected S"
 unfolding limitComplete_def joinClosed_def2 upwardsDirected_def3 lub_def least_def func_defs comb_defs by metis
lemma meetClosed_downwardsDirected: "limitComplete R ⟹ R-meetClosed S ⟹ R-downwardsDirected S"
   by (simp add: downwardsDirected_defT joinClosed_upwardsDirected limitComplete_defT meetClosed_defT)

text ‹Thus we have:›
lemma greatestExist_upwardsDirected: "R-greatestExist S ⟹ R-upwardsDirected S" 
  by (simp add: greatestExist_def2 upwardsDirected_def3)
lemma leastExist_downwardsDirected: "R-leastExist S ⟹ R-downwardsDirected S" 
  by (simp add: downwardsDirected_defT greatestExist_upwardsDirected leastExist_defT)
text ‹Note, however:›

text ‹The following notions abstract the order-theoretical property of filter/ideal towards relations in
 general: S is a filter/ideal when all and only pairs of S-elements have their meet/join (wrt R) in S.›

  where "R-filter S ≡ R-pseudoFilter S ∧ R-meetClosed S"

  where "R-ideal S  ≡ R-pseudoIdeal S ∧ R-joinClosed S"
